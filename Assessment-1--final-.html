total_fat_per_restaurant <- fastfood %>%
  group_by(restaurant) %>%
  summarise(total_fat = sum(total_fat)) %>%
  ungroup() %>%
  arrange(total_fat)

#Figure 1a and Figure 1b show the total fat per restaurant. Based on 2 variables which are one discrete and another is continuous, goem_col() function is chosen in Figure 1a, and geom_point function with geom_line is chosen in Figure 1b. In addition, text label and reorder functions are used.

total_fat_per_restaurant %>%
  mutate(restaurant = fct_reorder(restaurant, total_fat)) %>%
  ggplot(aes(restaurant, total_fat, group = 1, fill = total_fat)) +
  geom_col() +
  scale_fill_viridis_b() +
  coord_flip() +
  geom_text(aes(label = total_fat), fontface = "bold", size = 4, vjust = 0) +
  labs(x = NULL, y = "Total fat", 
       title = "TOTAL FAT OF EACH RESTAURANT", caption = "Figure 1a")

total_fat_per_restaurant %>%
  mutate(restaurant = fct_reorder(restaurant, total_fat)) %>%

  geom_point() +
      coord_flip() +

    labs(x = NULL, y = "Total fat",
